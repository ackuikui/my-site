## 6.质数因子（递归+最小公因子）
输入：180
输出：2 2 3 3 5

```js
// 递归，查找2到平方根 能被整除的的数
let arr = []
function getMin(num){
    let max = Math.floor(Math.sqrt(num))
    for(let i = 2; i <= max; i++){
        if(num % i == 0){
            num = num / i
            arr.push(i)
            getMin(num)
            // console.log(i)
            break;
        } else if(i == max) {
            arr.push(num)
        }
    }
}
```
